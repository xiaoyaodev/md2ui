## Python

```python
from dataclasses import dataclass
from typing import List, Optional

@dataclass
class Document:
    title: str
    content: str
    tags: List[str]
    parent: Optional['Document'] = None

    def render(self) -> str:
        return f"# {self.title}\n\n{self.content}"

    def word_count(self) -> int:
        return len(self.content.split())
```
